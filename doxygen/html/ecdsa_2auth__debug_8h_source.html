49 #ifdef NDEBUG
50 #define debug(M, ...)

52 #define debug(M, ...) \
53  dlog_printf(stderr, \
54  "[" BOLDBLUE "DEBUG" RESET \

60 #define clean_errno() (errno == 0 ? "None" : strerror(errno))
61 
62 #define log_err(M, ...) \
63  fprintf(stderr, \
64  "[" BOLDRED "ERROR" RESET \
65  "]" \
66  "(%s:%d: errno: %s) " M "\n", \
67  __FILE__, __LINE__, clean_errno(), ##__VA_ARGS__)
68 
69 #define log_warn(M, ...) \
70  fprintf(stderr, \
71  "[" BOLDMAGENTA "WARN" RESET \
72  "]" \
73  "(%s:%d: errno:%s) " M "\n", \
74  __FILE__, __LINE__, clean_errno(), ##__VA_ARGS__)
75 
76 #define log_info(M, ...) \
